def get_title_format(workbook):
  title_format = workbook.add_format({
    'bold': True,
    'bg_color': '#B4C7E7',
    'align': 'center',
    'valign': 'vcenter',
    'font_size': 12,
    'bottom': 1,
    'right': 1,
  })
  return title_format


def setup_first_sheet(workbook, worksheet):
